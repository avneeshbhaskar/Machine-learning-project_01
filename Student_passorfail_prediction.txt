import numpy as np
import pandas as pd
from sklearn.linear_model import LogisticRegression

# Sample dataset
data = {
    'Hours': [1, 2, 3, 4, 5, 6, 7, 8],
    'Pass':  [0, 0, 0, 0, 1, 1, 1, 1]
}

df = pd.DataFrame(data)

# Features and target
X = df[['Hours']]
y = df['Pass']

# Model
model = LogisticRegression()
model.fit(X, y)

# Prediction
hours = 3.5
prediction = model.predict([[hours]])

if prediction[0] == 1:
    print(f"Student will PASS if studies {hours} hours")
else:
    print(f"Student will FAIL if studies {hours} hours")
